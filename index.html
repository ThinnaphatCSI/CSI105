Why Study Big O?

Comprehensive Code Examples

⚡ O(1) - Constant Time (T(n) = c)

📐 Mathematical: T(n) = 1 (constant) 
// Array access - constant time
function getElement(arr, index) {
    return arr[index];  // Always takes the same time regardless of array size
}

// Mathematical operations
function add(a, b) {
    return a + b;  // Single operation - O(1)
}

// Hash table operations
const map = new Map();
map.set('key', 'value');  // O(1) average case
map.get('key');           // O(1) average case
Explanation: The execution time remains constant regardless of input size (n). Even with 1 billion elements, it still takes the same time.

📉 O(log n) - Logarithmic Time (T(n) = log₂ n)

📐 Mathematical: T(n) = log₂(n) (halves the search space each step) 
function binarySearch(arr, target) {
    let left = 0, right = arr.length - 1;
    
    while (left <= right) {
        const mid = Math.floor((left + right) / 2);
        if (arr[mid] === target) return mid;
        if (arr[mid] < target) left = mid + 1;
        else right = mid - 1;
    }
    return -1;
}
// Each iteration eliminates half of the remaining elements
// For n=1,000,000 → only about 20 iterations needed!
Explanation: The algorithm divides the problem in half each time. When n doubles, only one extra step is needed. Very efficient for large datasets!

📊 O(n) - Linear Time (T(n) = n)

📐 Mathematical: T(n) = n (linear relationship with input size) 
// Find maximum value
function findMax(arr) {
    let max = arr[0];
    for (let i = 1; i < arr.length; i++) {
        if (arr[i] > max) max = arr[i];  // Checks each element once
    }
    return max;
}

// Calculate sum
function sumArray(arr) {
    let sum = 0;
    for (let num of arr) sum += num;  // One pass through all elements
    return sum;
}

// Linear search
function linearSearch(arr, target) {
    for (let i = 0; i < arr.length; i++) {
        if (arr[i] === target) return i;  // Worst case: checks all n elements
    }
    return -1;
}
Explanation: Execution time grows proportionally with input size. If n doubles, execution time roughly doubles.

🚀 O(n log n) - Linearithmic Time (T(n) = n log₂ n)

📐 Mathematical: T(n) = n × log₂(n) (divide and conquer with linear merge) 
function mergeSort(arr) {
    if (arr.length <= 1) return arr;
    
    const mid = Math.floor(arr.length / 2);
    const left = mergeSort(arr.slice(0, mid));   // Divide into halves
    const right = mergeSort(arr.slice(mid));      // Recursively sort
    
    return merge(left, right);  // Merge takes O(n) time
}

function merge(left, right) {
    const result = [];
    let i = 0, j = 0;
    
    while (i < left.length && j < right.length) {
        if (left[i] < right[j]) result.push(left[i++]);
        else result.push(right[j++]);
    }
    return [...result, ...left.slice(i), ...right.slice(j)];
}

// Example: For n = 1,000,000 → about 20 million operations (n log n)
Explanation: This is the optimal complexity for comparison-based sorting. Much faster than O(n²) for large datasets.

🐌 O(n²) - Quadratic Time (T(n) = n²)

📐 Mathematical: T(n) = n² (nested iterations) 
// Find duplicate pairs
function findDuplicates(arr) {
    const duplicates = [];
    for (let i = 0; i < arr.length; i++) {
        for (let j = i + 1; j < arr.length; j++) {
            if (arr[i] === arr[j]) {
                duplicates.push(arr[i]);  // Compares every pair
            }
        }
    }
    return duplicates;
}

// Bubble sort
function bubbleSort(arr) {
    const n = arr.length;
    for (let i = 0; i < n - 1; i++) {
        for (let j = 0; j < n - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]];
            }
        }
    }
    return arr;
}

// Example: For n = 10,000 → 100 million operations!
Explanation: Nested loops cause quadratic growth. When n doubles, execution time quadruples. Avoid for large datasets!

🐢 O(n³) - Cubic Time (T(n) = n³)

📐 Mathematical: T(n) = n³ (triple nested loops) 
// Naive matrix multiplication (n × n matrices)
function matrixMultiply(A, B) {
    const n = A.length;
    const C = Array(n).fill().map(() => Array(n).fill(0));
    
    for (let i = 0; i < n; i++) {           // n iterations
        for (let j = 0; j < n; j++) {       // n iterations
            for (let k = 0; k < n; k++) {   // n iterations
                C[i][j] += A[i][k] * B[k][j];
            }
        }
    }
    return C;
}
// Example: For n = 1,000 → 1 billion operations!
Explanation: Triple nested loops cause cubic growth. Extremely slow for large inputs. Requires optimization techniques.

💀 O(2ⁿ) - Exponential Time (T(n) = 2ⁿ)

📐 Mathematical: T(n) = 2ⁿ (each call creates 2 recursive calls) 
// Recursive Fibonacci - Very inefficient
function fibonacci(n) {
    if (n <= 1) return n;
    return fibonacci(n - 1) + fibonacci(n - 2);
}
// Time: O(2ⁿ), Space: O(n)
// For n=50 → about 1.1 quadrillion operations!

// Better solution: O(n) using memoization or iteration
function fibonacciOptimized(n) {
    if (n <= 1) return n;
    let prev = 0, curr = 1;
    for (let i = 2; i <= n; i++) {
        [prev, curr] = [curr, prev + curr];
    }
    return curr;
}
Explanation: Exponential growth is extremely dangerous! Even n=50 can take years to compute. Always look for optimization!

💀💀 O(n!) - Factorial Time (T(n) = n!)

📐 Mathematical: T(n) = n! (all permutations) 
// Generate all permutations of a string
function permutations(str) {
    const result = [];
    
    function permute(current, remaining) {
        if (remaining.length === 0) {
            result.push(current);
            return;
        }
        for (let i = 0; i < remaining.length; i++) {
            permute(current + remaining[i], 
                    remaining.slice(0, i) + remaining.slice(i + 1));
        }
    }
    
    permute('', str);
    return result;
}
// For n=10 → 3,628,800 permutations
// For n=20 → 2.4×10¹⁸ permutations (impossible to compute)
Explanation: Factorial is the worst complexity. Only feasible for very small n (n ≤ 10). Used in traveling salesman problem brute force.

📐 Big O Calculation Rules
Rule 1: Drop ConstantsO(2n) → O(n), O(100) → O(1)
Rule 2: Keep Dominant TermO(n² + n) → O(n²)
Rule 3: Addition (Sequential)O(f) + O(g) = O(max(f,g))
